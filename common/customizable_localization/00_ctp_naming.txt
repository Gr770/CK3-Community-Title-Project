TierCheck = {
	type = character
	text = {
		trigger = {
			OR = {
				is_landed = yes
				AND = {
					exists = primary_spouse
					root.primary_spouse = {
						is_landed = yes
					}
				}
				AND = {
					exists = father
					root.father = {
						is_landed = yes
						highest_held_title_tier >= tier_kingdom
					}
				}
				AND = {
					exists = mother
					root.mother = {
						is_landed = yes
						highest_held_title_tier >= tier_kingdom
					}
				}
			}
		}
		localization_key = CTP_Tier
	}
	text = {
		fallback = yes
		localization_key = Gui_blank
	}
}
TekCheck = {
